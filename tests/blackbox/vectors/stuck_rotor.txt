# Stuck rotor detection via interval timer timeout
# Covers: REQ-MOTOR-STOP_DETECTION, REQ-PROT-STUCK_ROTOR
#
# When interval timer exceeds 45000 (22.5ms without BEMF zero-cross),
# bemf_timeout_happened increments. After exceeding bemf_timeout threshold
# (10 at high throttle), stuck_rotor_protection latches input=0, running=0.
#
# Note: interval_timer must be set EACH tick (zcfoundroutine resets it).
# C latches at tick 13, Rust at tick 14 (1-tick atomic publish delay).

@config
armed=1
inputSet=1
dshot=1
EDT_ARMED=1
running=1
old_routine=1
eeprom.stuck_rotor_protection=1

@sequence
# Normal: low interval timer, motor runs
tick 1  | throttle=500 interval_timer=1000 | running=1 input=500

# Sustain stall for 20 ticks
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |
tick 1  | throttle=500 interval_timer=50000 |

# After sustained stall: motor stopped, input latched to 0
tick 1  | throttle=500 interval_timer=50000 | input=0 running=0
